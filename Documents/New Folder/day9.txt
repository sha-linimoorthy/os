#include<stdio.h>
int main()
{
    int i,j,n;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        if(i==(n-1))
        {
            for(j=1;j<2*n+1;j++)
            {
                printf("%c",'*');
            }
            printf("\n");
        }
        else
        {
            for(j=2;j<2*n+1;j++)
            {
                if(j<=(n-i) || j>=(2*n)-(n-i))
                {
                    if(j==(n-i) || j==(2*n)-(n-i))
                    {
                        printf("*");
                    }
                    else
                    {
                        printf("b");
                    }
                }
                else
                {
                    printf("i");
                }
            }
            printf("\n");
        }
    }
}
